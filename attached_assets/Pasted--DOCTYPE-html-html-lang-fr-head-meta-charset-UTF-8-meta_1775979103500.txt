<script>
  const SUPABASE_URL = "https://bxpatupsirntpwkkrjas.supabase.co";
  const ANON_KEY = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6ImJ4cGF0dXBzaXJudHB3a2tyamFzIiwicm9sZSI6ImFub24iLCJpYXQiOjE3NzU5NDEzMjAsImV4cCI6MjA5MTUxNzMyMH0.cDkJIBYIdNJHhmQqQVJbiooO17ERBQ9z0Icf7zaXyFg";

  let token = null;
  let userId = null;
  let currentFilter = "toutes";
  let allTasks = [];

  function showMessage(msg, color = "green") {
    const el = document.getElementById("message");
    el.style.display = "block";
    el.style.background = color === "green" ? "#f0fff4" : "#fff5f5";
    el.style.color = color === "green" ? "#276749" : "#c53030";
    el.textContent = msg;
    setTimeout(() => el.style.display = "none", 3000);
  }

  async function signup() {
    const email = document.getElementById("email").value;
    const password = document.getElementById("password").value;
    const res = await fetch(`${SUPABASE_URL}/auth/v1/signup`, {
      method: "POST",
      headers: { "apikey": ANON_KEY, "Content-Type": "application/json" },
      body: JSON.stringify({ email, password })
    });
    const data = await res.json();
    if (data.access_token) {
      token = data.access_token;
      userId = data.user.id;
      showMessage("Inscription réussie !");
      showApp();
      loadTasks();
    } else {
      showMessage(data.msg || "Erreur", "red");
    }
  }

  async function login() {
    const email = document.getElementById("email").value;
    const password = document.getElementById("password").value;
    const res = await fetch(`${SUPABASE_URL}/auth/v1/token?grant_type=password`, {
      method: "POST",
      headers: { "apikey": ANON_KEY, "Content-Type": "application/json" },
      body: JSON.stringify({ email, password })
    });
    const data = await res.json();
    if (data.access_token) {
      token = data.access_token;
      userId = data.user.id;
      showMessage("Connecté !");
      showApp();
      loadTasks();
    } else {
      showMessage(data.msg || "Identifiants incorrects", "red");
    }
  }

  function logout() {
    token = null; userId = null;
    document.getElementById("auth-section").style.display = "block";
    document.getElementById("app-section").style.display = "none";
  }

  function showApp() {
    document.getElementById("auth-section").style.display = "none";
    document.getElementById("app-section").style.display = "block";
  }

  function setFilter(filter, btn) {
    currentFilter = filter;
    document.querySelectorAll(".filter-btn").forEach(b => b.classList.remove("active"));
    btn.classList.add("active");
    renderTasks();
  }

  async function loadTasks() {
    const res = await fetch(`${SUPABASE_URL}/rest/v1/tasks?order=start_time.asc`, {
      headers: { "apikey": ANON_KEY, "Authorization": `Bearer ${token}` }
    });
    allTasks = await res.json();
    updateStats();
    renderTasks();
  }

  function updateStats() {
    const total = allTasks.length;
    const done = allTasks.filter(t => t.done).length;
    const pending = total - done;
    document.getElementById("stat-total").textContent = total;
    document.getElementById("stat-done").textContent = done;
    document.getElementById("stat-pending").textContent = pending;
  }

  function renderTasks() {
    let filtered = allTasks;
    if (currentFilter === "done") filtered = allTasks.filter(t => t.done);
    if (currentFilter === "pending") filtered = allTasks.filter(t => !t.done);

    const list = document.getElementById("tasks-list");
    if (filtered.length === 0) {
      list.innerHTML = `<div class="empty">Aucune tâche ici</div>`;
      return;
    }

    list.innerHTML = filtered.map(t => `
      <div class="task-item ${t.done ? 'done' : 'pending'}">
        <div class="task-header">
          <span class="task-title ${t.done ? 'done-text' : ''}">${t.title}</span>
          <div class="task-actions">
            <button class="${t.done ? 'btn-gray' : 'btn-success'}" onclick="toggleTask('${t.id}', ${t.done})">
              ${t.done ? '↩️ Annuler' : '✅ Fait'}
            </button>
            <button class="btn-danger" onclick="deleteTask('${t.id}')">🗑️</button>
          </div>
        </div>
        <div class="task-meta">
          ${t.start_time ? `<span class="badge badge-time">⏰ ${t.start_time} - ${t.end_time || '?'}</span>` : ''}
          ${t.category ? `<span class="badge badge-category">📌 ${t.category}</span>` : ''}
          ${t.priority ? `<span class="badge badge-priority-${t.priority}">
            ${t.priority === 'haute' ? '🔴' : t.priority === 'basse' ? '🟡' : '🟢'} ${t.priority}
          </span>` : ''}
        </div>
      </div>
    `).join("");
  }

  async function createTask() {
    const title = document.getElementById("new-task").value.trim();
    if (!title) return showMessage("Titre requis", "red");
    const start_time = document.getElementById("start-time").value || null;
    const end_time = document.getElementById("end-time").value || null;
    const category = document.getElementById("category").value;
    const priority = document.getElementById("priority").value;

    await fetch(`${SUPABASE_URL}/rest/v1/tasks`, {
      method: "POST",
      headers: { "apikey": ANON_KEY, "Authorization": `Bearer ${token}`, "Content-Type": "application/json" },
      body: JSON.stringify({ title, start_time, end_time, category, priority, user_id: userId })
    });
    document.getElementById("new-task").value = "";
    document.getElementById("start-time").value = "";
    document.getElementById("end-time").value = "";
    showMessage("Tâche ajoutée !");
    loadTasks();
  }

  async function toggleTask(id, done) {
    await fetch(`${SUPABASE_URL}/rest/v1/tasks?id=eq.${id}`, {
      method: "PATCH",
      headers: { "apikey": ANON_KEY, "Authorization": `Bearer ${token}`, "Content-Type": "application/json" },
      body: JSON.stringify({ done: !done })
    });
    loadTasks();
  }

  async function deleteTask(id) {
    if (!confirm("Supprimer cette tâche ?")) return;
    await fetch(`${SUPABASE_URL}/rest/v1/tasks?id=eq.${id}`, {
      method: "DELETE",
      headers: { "apikey": ANON_KEY, "Authorization": `Bearer ${token}` }
    });
    loadTasks();
  }
